### Usage Example
```jsx
import SpectraSlider from './SpectraSlider'

const MY_CARDS = [
  { id: 1, image: 'https://example.com/photo1.jpg', label: 'Photo One' },
  { id: 2, image: 'https://example.com/photo2.jpg', label: 'Photo Two' },
  { id: 3, image: 'https://example.com/photo3.jpg', content: <div>Custom overlay</div> },
  // ...
];

<div style={{ width: '100%', height: '500px' }}>
  <SpectraSlider
    cards={MY_CARDS}
    numberOfCards={7}
    cardWidth={220}
    cardHeight={340}
    gap={190}
    perspective={1200}
    maxRotation={35}
    scaleStep={0.14}
    opacityStep={0.28}
    brightnessStep={0.35}
    draggable={true}
    snap={true}
    initialIndex={0}
    className=""
    cardClassName=""
    onCardClick={(card, index) => console.log(card, index)}
    autoPlay={false}
    duration={20}
    direction="vertical"
    curveIntensity={-20}
    reverse={false}
    pauseOnHover={true}
    mouseWheel={true}
    wheelSmoothness={0.15}

  />
</div>
```

### Card Data Shape
| Field | Type | Description |
|-------|------|-------------|
| id | string \| number | Unique key for React reconciliation. Falls back to array index. |
| image | string | URL of the card image. |
| label | string | Alt text for the image (defaults to empty string). |
| content | ReactNode | Optional JSX rendered as an absolute overlay at the bottom of the card (z-index 10). |

### Props
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| cards | array | FALLBACK_CARDS (built-in) | Array of card objects. Falls back to 7 built-in Unsplash images if empty. |
| numberOfCards | number | 7 | How many cards to render. Minimum 3, no upper cap (sliced from `cards`). |
| cardWidth | number | 220 | Width of each card in pixels. |
| cardHeight | number | 340 | Height of each card in pixels. |
| gap | number | 190 | Distance in pixels between adjacent cards along the layout axis. Also used as drag sensitivity denominator. |
| perspective | number | 1200 | CSS `perspective` value in pixels applied to the stage. Higher = flatter 3D depth. |
| maxRotation | number | 35 | Maximum rotation in degrees applied to off-center cards. |
| scaleStep | number | 0.14 | Scale reduction per card away from center (1 - absDiff × scaleStep). |
| opacityStep | number | 0.28 | Opacity reduction per card away from center. |
| brightnessStep | number | 0.35 | CSS brightness reduction per card away from center. |
| draggable | boolean | true | Enables drag and touch-swipe to navigate. |
| snap | boolean | true | Snaps to the nearest whole card index on drag/wheel release. |
| initialIndex | number | 0 | Card index shown at center on mount. |
| className | string | "" | Additional CSS classes on the outer stage div. |
| cardClassName | string | "" | Additional CSS classes on each card div. |
| onCardClick | function | undefined | Callback fired when the centered card is clicked: `(card, index) => void`. Clicking a non-center card navigates to it instead. |
| autoPlay | boolean | false | Auto-advances the carousel continuously. |
| duration | number | 20 | Time in seconds for one full carousel cycle when `autoPlay` is true. |
| direction | string | "vertical" | Layout and drag axis. Options: "vertical" \| "horizontal" |
| curveIntensity | number | -20 | Lateral curve offset applied to off-center cards (negative = curve inward). Set to 0 for a flat line. |
| reverse | boolean | false | Reverses the auto-play direction. |
| pauseOnHover | boolean | true | Pauses auto-play when the mouse enters any card. |
| mouseWheel | boolean | true | Enables scroll wheel to navigate (prevents default page scroll while hovering the component). |
| wheelSmoothness | number | 0.15 | Lerp factor for wheel-driven scroll animation (0 = no movement, 1 = instant). |
| wheelSensitivity | number | 0.008 | Multiplier applied to `deltaY` for wheel offset change. |
| filterOnHover | boolean | false | Toggles grayscale filter on cards on hover. |
| defaultFiltered | boolean | false | If true, all cards start grayscale and clear on hover. If false, all start unfiltered and grey on hover. |
| entranceAnimation | boolean | true | Plays a GSAP entrance animation on mount. |
| entranceDuration | number | 0.9 | Duration in seconds of the entrance animation per card. |
| entranceStagger | number | 0.07 | Delay in seconds between each card's entrance start. |
| entranceEase | string | "back.out(1.4)" | GSAP ease string for the entrance animation. |
| entranceType | string | "spin" | Entrance animation style. Options: "spin" \| "fade" \| "bottom" \| "top" \| "left" \| "right" \| "flip" \| "zoomBlur" \| "scatter" |

### Full Component Source
```jsx
// JS-TW variant
import { useCallback, useEffect, useRef, useState } from 'react';
import gsap from 'gsap';

const FALLBACK_CARDS = [
  { id: 1, image: 'https://images.unsplash.com/photo-1516979187457-637abb4f9353?w=500&q=70' },
  { id: 2, image: 'https://images.unsplash.com/photo-1465101162946-4377e57745c3?w=500&q=70' },
  { id: 3, image: 'https://images.unsplash.com/photo-1507003211169-0a1dd7228f2d?w=500&q=70' },
  { id: 4, image: 'https://images.unsplash.com/photo-1494548162494-384bba4ab999?w=500&q=70' },
  { id: 5, image: 'https://images.unsplash.com/photo-1500534314209-a25ddb2bd429?w=500&q=70' },
  { id: 6, image: 'https://images.unsplash.com/photo-1504198458649-3128b932f49e?w=500&q=70' },
  { id: 7, image: 'https://images.unsplash.com/photo-1519681393784-d120267933ba?w=500&q=70' },
];

const FILTER_STYLE = 'grayscale(100%) brightness(0.52) contrast(1.18)';

export function SpectraSlider({
  cards = [], numberOfCards = 7, cardWidth = 220, cardHeight = 340, gap = 190,
  perspective = 1200, maxRotation = 35, scaleStep = 0.14, opacityStep = 0.28,
  brightnessStep = 0.35, draggable = true, snap = true, initialIndex = 0,
  className = '', cardClassName = '', onCardClick, autoPlay = false, duration = 20,
  direction = 'vertical', curveIntensity = -20, reverse = false, pauseOnHover = true,
  mouseWheel = true, wheelSmoothness = 0.15, wheelSensitivity = 0.008,
  filterOnHover = false, defaultFiltered = false, entranceAnimation = true,
  entranceDuration = 0.9, entranceStagger = 0.07, entranceEase = 'back.out(1.4)',
  entranceType = 'spin',
}) {
  const sourceCards = cards.length > 0 ? cards : FALLBACK_CARDS;
  const activeCards = sourceCards.slice(0, Math.max(numberOfCards, 3));
  const total = activeCards.length;
  const isVertical = direction === 'vertical';

  const stageRef = useRef(null), cardRefs = useRef([]);
  const offsetRef = useRef(initialIndex);
  const dragRef = useRef({ active: false, startX: 0, startY: 0, startOffset: 0 });
  const tickerFnRef = useRef(null), tickingRef = useRef(false);
  const hoverPausedRef = useRef(false), hoveredRef = useRef(false), mountedRef = useRef(false);
  const wheelTargetRef = useRef(initialIndex), wheelTickerFnRef = useRef(null);
  const wheelTickingRef = useRef(false), wheelSnapTimerRef = useRef(null);

  const [filteredCards, setFilteredCards] = useState(() =>
    defaultFiltered ? new Set(activeCards.map((_, i) => i)) : new Set()
  );

  const totalRef = useRef(total), gapRef = useRef(gap), maxRotationRef = useRef(maxRotation);
  const scaleStepRef = useRef(scaleStep), opacityStepRef = useRef(opacityStep);
  const brightnessStepRef = useRef(brightnessStep), durationRef = useRef(duration);
  const reverseRef = useRef(reverse), autoPlayRef = useRef(autoPlay), directionRef = useRef(direction);
  const curveIntensityRef = useRef(curveIntensity), wheelSmoothnessRef = useRef(wheelSmoothness);
  const wheelSensitivityRef = useRef(wheelSensitivity);

  useEffect(() => {
    totalRef.current = total; gapRef.current = gap; maxRotationRef.current = maxRotation;
    scaleStepRef.current = scaleStep; opacityStepRef.current = opacityStep;
    brightnessStepRef.current = brightnessStep; durationRef.current = duration;
    reverseRef.current = reverse; autoPlayRef.current = autoPlay; directionRef.current = direction;
    curveIntensityRef.current = curveIntensity; wheelSmoothnessRef.current = wheelSmoothness;
    wheelSensitivityRef.current = wheelSensitivity;
  }, [total, gap, maxRotation, scaleStep, opacityStep, brightnessStep, duration, reverse, autoPlay, direction, curveIntensity, wheelSmoothness, wheelSensitivity]);

  const wrapDiff = (i, offset, total) => {
    let diff = (i - offset) % total;
    if (diff > total / 2) diff -= total;
    if (diff < -total / 2) diff += total;
    return diff;
  };

  const placeCards = useCallback(() => {
    const vertical = directionRef.current === 'vertical';
    cardRefs.current.forEach((el, i) => {
      if (!el) return;
      const diff = wrapDiff(i, offsetRef.current, totalRef.current), absDiff = Math.abs(diff);
      const offset = diff * gapRef.current, curveOffset = curveIntensityRef.current * absDiff * absDiff;
      const rotate = gsap.utils.clamp(-maxRotationRef.current, maxRotationRef.current, -diff * 22);
      const scale = Math.max(0.35, 1 - absDiff * scaleStepRef.current);
      const opacity = Math.max(0, 1 - absDiff * opacityStepRef.current);
      const brightness = Math.max(0.25, 1 - absDiff * brightnessStepRef.current);
      gsap.set(el, {
        x: vertical ? curveOffset : offset,
        y: vertical ? offset : curveOffset,
        z: -absDiff * 60,
        rotateY: vertical ? 0 : rotate,
        rotateX: vertical ? rotate : 0,
        scale, opacity,
        zIndex: Math.round(100 - absDiff * 10),
        filter: `brightness(${brightness})`,
      });
    });
  });

  const stopTicker = () => {
    if (!tickingRef.current || !tickerFnRef.current) return;
    gsap.ticker.remove(tickerFnRef.current); tickerFnRef.current = null; tickingRef.current = false;
  };
  const startTicker = () => {
    if (tickingRef.current) return;
    const dir = reverseRef.current ? -1 : 1;
    tickerFnRef.current = () => {
      offsetRef.current += (dir * totalRef.current) / (durationRef.current * 60);
      offsetRef.current = ((offsetRef.current % totalRef.current) + totalRef.current) % totalRef.current;
      placeCards();
    };
    gsap.ticker.add(tickerFnRef.current); tickingRef.current = true;
  };
  const maybeStartTicker = () => {
    if (autoPlayRef.current && !hoverPausedRef.current && !dragRef.current.active && !wheelTickingRef.current)
      startTicker();
  };

  const getEntranceVars = type => ({
    fade:     { opacity: 0 },
    bottom:   { opacity: 0, y: '+=140' },
    top:      { opacity: 0, y: '-=140' },
    left:     { opacity: 0, x: '-=180' },
    right:    { opacity: 0, x: '+=180' },
    flip:     { opacity: 0, scaleX: 0 },
    spin:     { opacity: 0, scale: 0, rotation: '+=360' },
    zoomBlur: { opacity: 0, scale: 2.4, filter: 'blur(14px)' },
    scatter:  { opacity: 0, scale: 0.3, x: () => gsap.utils.random(-300, 300), y: () => gsap.utils.random(-300, 300), rotation: () => gsap.utils.random(-180, 180) },
  }[type] ?? { opacity: 0, scale: 0, rotation: 0 });

  useEffect(() => {
    placeCards();
    if (!entranceAnimation) { mountedRef.current = true; maybeStartTicker(); return; }
    const validEls = cardRefs.current.filter(Boolean);
    if (!validEls.length) return;
    gsap.from(validEls, {
      ...getEntranceVars(entranceType),
      duration: entranceDuration, ease: entranceEase, stagger: entranceStagger, overwrite: 'auto',
      onComplete: () => { mountedRef.current = true; maybeStartTicker(); },
    });
  }, []);

  useEffect(() => {
    if (!mountedRef.current) return;
    if (autoPlay && !hoverPausedRef.current && !dragRef.current.active) startTicker(); else stopTicker();
    return () => stopTicker();
  }, [autoPlay, reverse, duration]);

  useEffect(() => {
    if (!mountedRef.current) return;
    placeCards();
  }, [total, gap, maxRotation, scaleStep, opacityStep, cardWidth, cardHeight, direction, placeCards]);

  const snapToNearest = () => {
    const target = Math.round(offsetRef.current);
    wheelTargetRef.current = target;
    gsap.to(offsetRef, {
      current: target, duration: 0.5, ease: 'power3.out', onUpdate: placeCards,
      onComplete: () => {
        offsetRef.current = ((offsetRef.current % totalRef.current) + totalRef.current) % totalRef.current;
        wheelTargetRef.current = offsetRef.current; placeCards(); maybeStartTicker();
      },
    });
  };

  const onCardPointerDown = e => {
    if (!draggable) return;
    dragRef.current = { active: true, startX: e.touches ? e.touches[0].clientX : e.clientX, startY: e.touches ? e.touches[0].clientY : e.clientY, startOffset: offsetRef.current };
    gsap.killTweensOf(offsetRef); stopTicker(); stopWheelTicker();
  };
  const onPointerMove = e => {
    if (!dragRef.current.active) return;
    const vertical = directionRef.current === 'vertical';
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    const delta = vertical ? clientY - dragRef.current.startY : clientX - dragRef.current.startX;
    offsetRef.current = dragRef.current.startOffset - delta / gapRef.current;
    wheelTargetRef.current = offsetRef.current; placeCards();
  };
  const onPointerUp = () => {
    if (!dragRef.current.active) return;
    dragRef.current.active = false;
    if (snap) snapToNearest(); else maybeStartTicker();
  };

  useEffect(() => {
    window.addEventListener('mousemove', onPointerMove); window.addEventListener('mouseup', onPointerUp);
    window.addEventListener('touchmove', onPointerMove, { passive: true }); window.addEventListener('touchend', onPointerUp);
    return () => {
      window.removeEventListener('mousemove', onPointerMove); window.removeEventListener('mouseup', onPointerUp);
      window.removeEventListener('touchmove', onPointerMove); window.removeEventListener('touchend', onPointerUp);
    };
  }, []);

  const stopWheelTicker = () => {
    if (!wheelTickingRef.current || !wheelTickerFnRef.current) return;
    gsap.ticker.remove(wheelTickerFnRef.current); wheelTickerFnRef.current = null; wheelTickingRef.current = false;
  };
  const startWheelTicker = () => {
    if (wheelTickingRef.current) return;
    wheelTickingRef.current = true;
    wheelTickerFnRef.current = () => {
      const diff = wheelTargetRef.current - offsetRef.current;
      if (Math.abs(diff) < 0.001) {
        offsetRef.current = wheelTargetRef.current; placeCards(); stopWheelTicker();
        if (snap) { clearTimeout(wheelSnapTimerRef.current); wheelSnapTimerRef.current = setTimeout(snapToNearest, 100); }
        else maybeStartTicker();
        return;
      }
      offsetRef.current += diff * wheelSmoothnessRef.current; placeCards();
    };
    gsap.ticker.add(wheelTickerFnRef.current);
  };

  useEffect(() => {
    const el = stageRef.current; if (!el || !mouseWheel) return;
    const handler = e => {
      if (!hoveredRef.current) return;
      e.preventDefault(); gsap.killTweensOf(offsetRef); stopTicker(); clearTimeout(wheelSnapTimerRef.current);
      if (!wheelTickingRef.current) wheelTargetRef.current = offsetRef.current;
      wheelTargetRef.current += e.deltaY * wheelSensitivityRef.current;
      startWheelTicker();
    };
    el.addEventListener('wheel', handler, { passive: false });
    return () => { el.removeEventListener('wheel', handler); stopWheelTicker(); clearTimeout(wheelSnapTimerRef.current); };
  }, [mouseWheel, snap]);

  const handleStageMouseEnter = () => { if (!pauseOnHover || !autoPlay) return; hoverPausedRef.current = true; stopTicker(); };
  const handleStageMouseLeave = () => { if (!pauseOnHover || !autoPlay) return; hoverPausedRef.current = false; if (!dragRef.current.active) startTicker(); };

  const handleCardClick = (card, index) => {
    const diff = wrapDiff(index, offsetRef.current, totalRef.current);
    if (Math.abs(diff) < 0.05) {
      onCardClick?.(card, index);
    } else {
      stopTicker(); stopWheelTicker();
      const target = offsetRef.current + diff;
      gsap.to(offsetRef, {
        current: target, duration: 0.5, ease: 'power3.out', onUpdate: placeCards,
        onComplete: () => {
          offsetRef.current = ((offsetRef.current % totalRef.current) + totalRef.current) % totalRef.current;
          wheelTargetRef.current = offsetRef.current; placeCards(); maybeStartTicker();
        },
      });
    }
  };

  const setCardFiltered = (index, value) =>
    setFilteredCards(prev => { const next = new Set(prev); value ? next.add(index) : next.delete(index); return next; });
  const handleCardMouseEnter = index => { hoveredRef.current = true; handleStageMouseEnter(); if (filterOnHover) setCardFiltered(index, !defaultFiltered); };
  const handleCardMouseLeave = index => { hoveredRef.current = false; handleStageMouseLeave(); if (filterOnHover) setCardFiltered(index, defaultFiltered); };

  const stageStyle = isVertical
    ? { height: '100%', minHeight: cardHeight + 60, perspective }
    : { height: cardHeight + 60, perspective };

  return (
    <div ref={stageRef} className={`relative w-full flex items-center justify-center select-none ${className}`} style={stageStyle}>
      <div className="relative" style={{ width: cardWidth, height: cardHeight, transformStyle: 'preserve-3d' }}>
        {activeCards.map((card, i) => (
          <div
            key={card.id ?? i}
            ref={el => (cardRefs.current[i] = el)}
            className={`absolute top-0 left-0 rounded-2xl overflow-hidden bg-neutral-900 ${cardClassName}`}
            style={{ width: cardWidth, height: cardHeight, willChange: 'transform' }}
            onClick={() => handleCardClick(card, i)}
            onMouseDown={onCardPointerDown}
            onTouchStart={onCardPointerDown}
            onMouseEnter={() => handleCardMouseEnter(i)}
            onMouseLeave={() => handleCardMouseLeave(i)}
          >
            {card.image && <img src={card.image} alt={card.label ?? ''} className="w-full h-full object-cover pointer-events-none" draggable={false} style={{ filter: filteredCards.has(i) ? FILTER_STYLE : 'none' }} />}
            {card.content && <div className="absolute inset-0 z-10 flex flex-col justify-end p-4">{card.content}</div>}
          </div>
        ))}
      </div>
    </div>
  );
}

export default SpectraSlider;
```

### Integration Instructions
1. Install required dependency: `npm install gsap`
2. Copy the component source above into `./SpectraSlider.jsx` in your project.
3. The stage is `w-full` horizontally. For `direction="vertical"`, give the parent a fixed height (e.g. `height: 500px`) — the stage uses `height: 100%` in vertical mode and will collapse without it. For `direction="horizontal"`, the stage sets its own height based on `cardHeight`.
4. All cards are stacked at `absolute top-0 left-0` inside a single fixed-size origin div. The 3D positioning is done entirely via `gsap.set` transforms — do not apply conflicting CSS transforms to cards.
5. The scroll wheel is only active when `hoveredRef.current` is true (i.e. the mouse is over a card). Moving the pointer out of the cards stops wheel capture and resumes normal page scroll.
6. Clicking a non-center card navigates to it (animates `offsetRef` to bring it to center) instead of firing `onCardClick`. `onCardClick` only fires when the clicked card is already at center (within 0.05 index units).
7. `wrapDiff` modulo arithmetic keeps card order continuous — the carousel loops seamlessly regardless of how far the offset scrolls.
8. `gap` controls both the visual spacing between cards and the drag sensitivity. Larger `gap` = cards are further apart and require more drag distance to advance one step.
9. Two independent GSAP ticker functions are used — one for auto-play (`tickerFnRef`) and one for smooth wheel lerp (`wheelTickerFnRef`). They never run simultaneously; each stops the other before starting.
10. `curveIntensity` applies a quadratic offset on the perpendicular axis (e.g. for `direction="vertical"`, it curves cards left/right). Set to `0` for a perfectly straight stack. Negative values curve inward, positive values curve outward.
